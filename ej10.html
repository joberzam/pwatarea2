function responder(opcion){ let correcta = preguntas[indice].correcta; let botones = document.querySelectorAll("#opciones button"); let resultado = document.getElementById("resultado"); // Deshabilitar botones después de responder botones.forEach(btn => btn.disabled = true); if(opcion === correcta){ puntaje++; botones[opcion].classList.add("correcto"); resultado.innerText = "Correcto ✅"; }else{ botones[opcion].classList.add("incorrecto"); botones[correcta].classList.add("correcto"); resultado.innerText = "Incorrecto ❌"; } // Pasar a la siguiente pregunta setTimeout(() => { indice++; if(indice < preguntas.length){ cargarPregunta(); }else{ mostrarResultado(); } }, 1500); }
